📋 完整的 router.go 示例代码
package router

import (
    "LeoAi/config"
    "LeoAi/internal/handler"
    "LeoAi/internal/middleware"
    "net/http"

    "github.com/gin-gonic/gin"
)

func SetupRouter(
    userHandler *handler.UserHandler,
    documentHandler *handler.DocumentHandler,
    taskHandler *handler.TaskHandler,
    cfg *config.Config,
) *gin.Engine {
    r := gin.Default()

    // ====================== 静态前端文件 ======================
    r.Static("/static", "./static")
    r.GET("/", func(c *gin.Context) {
        c.Redirect(http.StatusFound, "/app")
    })

    // ====================== 前端页面路由 ======================
    r.GET("/app", func(c *gin.Context) {
        c.File("./static/app.html")
    })
    r.GET("/auth", func(c *gin.Context) {
        c.File("./static/auth.html")
    })

    // ====================== API 路由 ======================
    // 公开 API（登录/注册无需认证）
    public := r.Group("/api/v1")
    {
        auth := public.Group("/auth")
        {
            auth.POST("/register", userHandler.Register)
            auth.POST("/login", userHandler.Login)
        }
    }

    // 需要 JWT 认证的 API
    protected := r.Group("/api/v1")
    protected.Use(middleware.JWTAuth(cfg))
    {
        // 文档管理
        doc := protected.Group("/documents")
        {
            doc.POST("", documentHandler.Upload)
            doc.GET("", documentHandler.List)
        }

        // AI 任务
        task := protected.Group("/tasks")
        {
            task.POST("", taskHandler.SubmitTask)
            task.GET("/:id", taskHandler.GetTaskStatus)
        }
    }

    return r
}
🗂️ 页面说明
app.html 主工作台 — 单页应用，通过侧边栏导航切换 7 个功能模块：概览、文档管理、AI 生成、任务中心、分类管理、标签管理、个人设置。未登录用户自动重定向到 /auth。
auth.html 登录 / 注册页面 — 支持登录和注册 Tab 切换，已登录用户自动跳转 /app。登录成功后 token 保存在 localStorage。
💡 设计说明
所有页面采用深色玻璃态设计风格（Glassmorphism），融入动态极光背景、流光渐变按钮、微妙的网格纹理。兼容 PC / 平板 / 手机三端自适应。

设计参考：Linear / Vercel / Notion AI — 年轻化、灵动、现代感强，科技感但不冷漠。
